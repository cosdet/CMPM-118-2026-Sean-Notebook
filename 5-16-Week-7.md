```rust
fn print_screen(stdout: &mut Stdout, editor: &mut Editor) {
    let file_buf = &editor.file_buffer;
    let (roff, coff) = (file_buf.row_offset, file_buf.col_offset);
    let (r, c) = (file_buf.row + 1, file_buf.col + 1);
    let (w, h) = termion::terminal_size().expect("Unsupported terminal."); // <-- ERROR HERE

    // Clear and start writing from origin
    write!(stdout, "{}{}", termion::clear::All, Goto(1, 1)).unwrap();

    for i in roff..(roff + h as usize - 1) {
        if i < file_buf.lines.len() {
            // Content
            let line = file_buf.lines.get(i).unwrap();

            if line.len() < coff {
                write!(stdout, "\n\r").unwrap();
                continue;
            }

            let part = &line[coff..min(coff + w as usize, line.len())];
            write!(stdout, "{}\n\r", to_str(&Vec::from(part))).unwrap();
        } else {
            // ~ as filler for parts of the window that are outside the buffer
            write!(stdout, "~\n\r").unwrap();
        }
    }

    // Status bar
    let (prompt, mut cmd, cmd_col) = editor.prompt();
    let bar_right = format!(" row: {}, col: {}", r, c);
    let avail = w as usize - bar_right.len() - prompt.len();
    let cmd_cur_pos = (prompt.len() + cmd_col + 1) as u16;
    let mut start = 0;

    let mut spacer = String::new();
    if cmd.len() > avail {
        start = min(cmd_col, cmd.len() - avail);
        cmd = cmd[start..(start + avail)].to_string();
    } else {
        spacer = " ".repeat(avail - cmd.len());
    }

    let bar = format!("{}{}{}{}", prompt, cmd, spacer, bar_right);
    write!(stdout, "{}{}{}", Invert, bar, Reset).unwrap();

    // Draw cursor on the right place
    match editor.state {
        State::Femto => write!(stdout, "{}", Goto((c - coff) as u16, (r - roff) as u16)).unwrap(),
        _ => write!(stdout, "{}", Goto(cmd_cur_pos - start as u16, h)).unwrap(),
    }
    // Ensure everything visible
    stdout.flush().unwrap();

    editor.message = None;
}
```

This function panics almost immediately at line `288` or upon calling `termion::terminal_size()`.

Looking into `src/ports/termion/src/sys/unix/size.rs` we can see the culprit of this panic:

```rust
use std::{io, mem};

use super::cvt;
use super::libc::{c_ushort, ioctl, STDOUT_FILENO, TIOCGWINSZ};

#[repr(C)]
struct TermSize {
    row: c_ushort,
    col: c_ushort,
    x: c_ushort,
    y: c_ushort,
}
/// Get the size of the terminal.
pub fn terminal_size() -> io::Result<(u16, u16)> {
    unsafe {
        let mut size: TermSize = mem::zeroed();
        cvt(ioctl(STDOUT_FILENO, TIOCGWINSZ.into(), &mut size as *mut _))?;
        Ok((size.col as u16, size.row as u16))
    }
}
```
</details>

It's `mlibc`'s `ioctl` and its mischevious henchman `TIOCGWINSZ`!

`ioctl()` seems to be the issue here as, in the most recent merge, `TIOCGWINSZ` is properly defined as `pub const TIOCGWINSZ: c_int = 0x5413;` in `src/ports/libc/src/twizzler/mod.rs`.

So, let's look at `ioctl()` and its header defintion
`src/ports/libc/src/twizzler/mod.rs` <- Header definition
`toolchain/src/mlibc/sysdeps/twizzler/sysdeps.cpp` <- Actual code

Seems like if the request is for `TIOCGWINSZ`, then the good old `fd_get_config()` function is called! I did some work on this previously, but the newest `main` changes change how file descriptors work, so my work is kind of moot. But I do know where to find this function and what it kind of should do!

Here is the latest version of `fd_get_config()` in `src/rt/reference/src/runtime/file.rs`

<details>
  <summary>fd_get_config()</summary>
  
  ```rust
  pub fn fd_get_config(
          &self,
          fd: RawFd,
          reg: u32,
          val: *mut c_void,
          val_len: usize,
      ) -> Result<()> {
          let mut binding = get_fd_slots().lock().unwrap();
          let Some(fd) = binding.get_mut(fd.try_into().unwrap()) else {
              return Err(TwzError::INVALID_ARGUMENT);
          };
  
          if reg == IO_REGISTER_IO_FLAGS {
              if val_len != size_of::<u32>() {
                  return Err(TwzError::INVALID_ARGUMENT);
              }
              unsafe { val.cast::<u32>().write(fd.flags.load(Ordering::SeqCst)) };
              return Ok(());
          }
  
          let buf = unsafe { core::slice::from_raw_parts_mut(val.cast::<u8>(), val_len) };
          buf.fill(0);
          fd.file.get_config(reg, val, val_len).into()
      }
  ```

</details>
